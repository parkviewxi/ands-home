CREATE TABLE tss_order (
  house_cd       VARCHAR(10)  NOT NULL,   -- 사업장
  com_cd        VARCHAR(10)  NOT NULL,   -- 회원사
  order_no      VARCHAR(20)  NOT NULL,   -- 주문번호 PK
  order_dt      DATE,
  status_cd     VARCHAR(4),
  cargo_wt      NUMERIC(12,3),
  del_yn        CHAR(1)      DEFAULT 'N',
  frst_usr_id  VARCHAR(20),
  frst_rgst_dttm TIMESTAMP,
  lst_chg_usr_id VARCHAR(20),
  lst_chg_dttm TIMESTAMP,
  PRIMARY KEY (house_cd, com_cd, order_no),
  FOREIGN KEY (com_cd) REFERENCES tpa020m(com_cd)
);
